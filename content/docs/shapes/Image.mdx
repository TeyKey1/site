To create an image with `Konva`, you can use the `Konva.Image()` object.

For full list of properties and methods, see the [Image API Reference](/api/Konva.Image.html).

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

<Tabs>
  <TabItem value="Vanilla" default>
```js live vanilla
import Konva from 'konva';

const stage = new Konva.Stage({
  container: 'container',
  width: window.innerWidth,
  height: window.innerHeight
});

const layer = new Konva.Layer();
stage.add(layer);

const imageObj = new Image();
imageObj.onload = function () {
  const konvaImage = new Konva.Image({
    x: 50,
    y: 50,
    image: imageObj,
    width: 100,
    height: 100
  });

  layer.add(konvaImage);
};
imageObj.src = 'path/to/image.jpg';
```

  <TabItem value="React">
```js live react
import { Stage, Layer, Image } from 'react-konva';
import { useEffect, useState } from 'react';

const App = () => {
  const [image, setImage] = useState(null);

  useEffect(() => {
    const img = new window.Image();
    img.src = 'path/to/image.jpg';
    img.onload = () => setImage(img);
  }, []);

  return (
    <Stage width={window.innerWidth} height={window.innerHeight}>
      <Layer>
        {image && <Image x={50} y={50} image={image} width={100} height={100} />}
      </Layer>
    </Stage>
  );
};

export default App;
```

```js live vue
<template>
  <v-stage :config="stageSize">
    <v-layer>
      <v-image :config="imageConfig" />
    </v-layer>
  </v-stage>
</template>

<script setup>
import { ref, onMounted } from 'vue';

const stageSize = {
  width: window.innerWidth,
  height: window.innerHeight
};

const imageConfig = ref({
  x: 50,
  y: 50,
  image: null,
  width: 100,
  height: 100
});

onMounted(() => {
  const img = new Image();
  img.src = 'path/to/image.jpg';
  img.onload = () => {
    imageConfig.value.image = img;
  };
});
</script>
```
